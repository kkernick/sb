50 template <class C> void resolve(const C& required, const std::string_view& program, const std::string& lib_cache, const std::string_view& hash) {
51 shared::log({"Resolving SOF"});
52
53 // Generate the list of invalid entries. Because
54 // we only read to the set, there is no risk in sharing it between
55 // the threads, so no mutex required.
56 shared::set exclusions = {};
57 for (const auto& [lib, mod] : arg::modlist("libraries")) {
58 if (mod == "x") {
59 if (lib.contains("*"))
60 exclusions.merge(shared::wildcard(lib, "/usr/lib", {"-maxdepth", "1", "-mindepth", "1", "-type", "f,l", "-executable"}));
61 else if (std::filesystem::is_directory(lib) && directories.contains(lib))
62 directories.erase(lib);
63 else exclusions.emplace(lib);
64 }
65 }
66
67 lib_t trimmed = {};
68 for (const auto& lib : required) {
69 if (!exclusions.contains(lib)) trimmed.emplace(lib);
70 }
71
72 libraries::setup(trimmed, program);
73 auto lib_out = std::ofstream(lib_cache);
74 lib_out << hash << '\n';
75 for (const auto& arg : trimmed) {
76 lib_out << arg << ' ';
77 }
78 lib_out.close();
79 }
