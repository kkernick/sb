84 bool digest_keypair(const std::string_view& key, const std::string_view& val, const uint_fast8_t& pos) {
85 // Single value modifiers are easy to split.
86 if (!list && custom == custom_policy::MODIFIABLE && val.contains(':')) {
87
88 // Split on colons.
89 auto s = shared::init<shared::vector>(shared::split, val, ':', false);
90 if (s.size() < 2) throw std::runtime_error("Invalid modifier: " + std::string(val));
91
92 // Grab the modifier keypair.
93 auto v = s[0]; auto mod = val.substr(val.find(':') + 1);
94
95 // If there are no valids, or it is valid, modify.
96 if (valid.size() <= 1 || valid.contains(v)) {
97 value = parser(v);
98 modifier = m_parser(mod);
99 return true;
100 }
101 }
102
103 // Ensure the key matches this argument
104 else if (key == long_name || key == short_name) {
105 if (long_name == "--help") throw std::runtime_error("Help!");
106
107 // Using ! sets value to their default values, and clears list.
108 if (val == "!") {
109 value = def;
110 modifier = "";
111 list_val.clear();
112 return true;
113 }
114
115 // Ensure the value is valid.
116 else if (valid.size() <= 1 || custom == custom_policy::TRUE || std::find(valid.begin(), valid.end(), val) != valid.end()) {
117
118 // If our key already has multiple values, split and handle separately.
119 if (list && val.contains(',')) {
120 for (const auto& x : shared::init<shared::vector>(shared::split, val, ',', false))
121 digest_keypair(key, x, pos);
122 }
123
124 // Otherwise just update.
125 else if (list) list_val.emplace(parser(val));
126 else value = parser(val);
127 return true;
128 }
129 }
130 return false;
131 }

205 bool digest(const shared::vector& args, uint_fast8_t& x) {
206
207 // Get the value
208 const auto& key = args[x];
209 auto ret = false;
210
211 // If a .sb script ends with a list argument, new args
212 // may be "consumed" by the list. However, we need to append
213 // arguments to the end for proper overriding. Scripts, therefore
214 // use "-- $@" to ensure that new arguments are not attached to a
215 // list
216 if (key == "--") return true;
217
218 // If we have an equal, it's `key=value`
219 if (key.contains("=")) {
220 auto keypair = shared::init<shared::vector>(shared::split, key, '=', false);
221 ret = digest_keypair(keypair[0], keypair[1], x);
222 }
223
224 // If we have a short name, digest each character.
225 // There is no support for values for short names.
226 else if (key[0] == '-' && key[1] != '-' && key.length() > 2) {
227 for (uint_fast8_t x = 1; x < key.length(); ++x) {
228 if (key[x] == short_name[1]) {
229 increment();
230 }
231 }
232 // We don't return true here because with a collection of args we
233 // need to parse for each match.
234 }
235
236 if (key == "--help") throw std::runtime_error("Help!");
237 if (key == "--version") throw std::runtime_error("Version");
238
239 // / Otherwise, if the next argument isn't a switch, we have `key value`
240 else if (!key.empty() && (key == long_name || key == short_name)) {
241 if (x + 1 != args.size() && !args[x].empty() && args[x + 1][0] != '-') {
242 if (list) {
243 while (++x < args.size() && !args[x].empty() && args[x][0] != '-')
244 digest_keypair(key, args[x], x);
245 ret = true;
246 --x;
247 }
248 else ret = digest_keypair(key, args[++x], x);
249 }
250
251 // If there are stages to the switch, increment it.
252 else if (valid.size() > 1) {
253 increment();
254 ret = true;
255 }
256 }
257
258 // If the position matches, and nothing else did, assume the key is the value.
259 else if (positional == x) {
260 if (key.starts_with("-"))
261 throw std::runtime_error("Invalid argument passed to: " + long_name);
262 value = parser(key);
263 ret = true;
264 }
265
266 // Update the set flag.
267 set |= ret;
268
269 if (x == positional && !ret)
270 throw std::runtime_error("Positional argument not set: " + long_name);
271 return ret;
272 }

289 std::string get_help() const {

291 help_text << long_name;

293 help_text << ' ';

303 help_text << v;
304 if (custom == custom_policy::MODIFIABLE) help_text << ":MOD";
305 if (v != *std::prev(order.end())) help_text << ',';
306 }
307 help_text << '}';
308 }
309 if (list) help_text << "...]";
310
311 help_text << "\n\t" << help << '\n';
