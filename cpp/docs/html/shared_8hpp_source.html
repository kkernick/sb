253 template <class R = void, class T = list> R exec(
254 const T& cmd,
255 const std::function<R(const int&, const int&)>& parser = [](const int& fd, const int& pid) {close(fd);},
256 const exec_return& policy = NONE
257 ) {
258
259 // Open a pipe for communication.
260 int pipefd[2];
261 if (pipe(pipefd) == -1) throw std::runtime_error("Failed to setup pipe");
262
263 auto pid = fork();
264 if (pid < 0) throw std::runtime_error("Failed to fork");
265 else if (pid == 0) {
266
267 // Zip the command, open our standard files to the pipe.
268 auto argv = zip(cmd);
269
270 close(pipefd[0]);
271 switch (policy) {
272 case STDOUT: dup2(pipefd[1], STDOUT_FILENO); break;
273 case STDERR: dup2(pipefd[1], STDERR_FILENO); break;
274 case ALL: dup2(pipefd[1], STDOUT_FILENO); dup2(pipefd[1], STDERR_FILENO); break;
275 default: break;
276 }
277
278 // Close the pipe and execute.
279 close(pipefd[1]);
280 execvp(argv[0], const_cast<char* const*>(argv.data()));
281 }
282
283 // Close the pipe and parser the other side.
284 close(pipefd[1]);
285 return parser(pipefd[0], pid);
286 }

458 template <class T, class A, class L = list, class ...Args> void batch(const A& fun, T& accum, const L& mem, const Args&... args) {
459 std::vector<std::future<T>> futures; futures.reserve(mem.size());
460 for (auto& val : mem) {
461 futures.emplace_back(pool.submit_task([val = std::move(val), &fun, &args...]() {return init<T>(fun, val, args...);}));
462 }
463 for (auto& future : futures) extend(accum, future.get());
464 }
