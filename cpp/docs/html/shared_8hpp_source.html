65 void generate(const std::string_view& parent, const std::string_view& prefix, const std::string_view& suffix) {
66 path.reserve(parent.length() + prefix.length() + suffix.length() + 17);
67 path.append(parent); if (!parent.ends_with('/')) path += '/';
68
69 std::stringstream name_stream;
70 if (!prefix.empty()) name_stream << prefix << '-';
71 name_stream << std::hex << rand(prng);
72 if (!suffix.empty()) name_stream << '-' << suffix;
73 name = name_stream.str();
74 path += name;
75 }

237 template <uint_fast8_t count> std::string head(const int& fd, const int& pid) {
238 std::array<char, count> buffer;
239 if (read(fd, buffer.data(), count) != count)
240 throw std::runtime_error("Failed to read file!");
241 close(fd);
242 if (pid > 0) kill(pid, SIGTERM);
243 return std::string(buffer.data(), count);

255 template <class R = void, class T = list> R exec(
256 const T& cmd,
257 const std::function<R(const int&, const int&)>& parser = [](const int& fd, const int& pid) {close(fd);},
258 const exec_policy& policy = NONE,
259 const std::string& in = ""
260 ) {
261
262 // Open a pipe for communication.
263 int output[2];
264 if (pipe(output) == -1) throw std::runtime_error("Failed to setup pipe");
265
266 int input[2] = {-1, -1};
267 if (!in.empty()) {
268 if (pipe(input) == -1) throw std::runtime_error("Failed to setup pipe");
269 }
270
271 auto pid = fork();
272 if (pid < 0) throw std::runtime_error("Failed to fork");
273 else if (pid == 0) {
274 // Zip the command, open our standard files to the pipe.
275 auto argv = zip(cmd);
276
277 close(output[0]);
278 switch (policy) {
279 case STDOUT: dup2(output[1], STDOUT_FILENO); break;
280 case STDERR: dup2(output[1], STDERR_FILENO); break;
281 case ALL: dup2(output[1], STDOUT_FILENO); dup2(output[1], STDERR_FILENO); break;
282 default: break;
283 }
284
285 if (!in.empty()) {
286 close(input[1]);
287 dup2(input[0], STDIN_FILENO);
288 close(input[1]);
289 }
290
291 // Close the pipe and execute.
292 close(output[1]);
293 execvp(argv[0], const_cast<char* const*>(argv.data()));
294 }
295
296 if (!in.empty()) {
297 close(input[0]);
298 write(input[1], in.c_str(), in.length());
299 close(input[1]);
300 }
301
302 // Close the pipe and parser the other side.
303 close(output[1]);
304 return parser(output[0], pid);
305 }

471 template <class T, class A, class L = list, class ...Args> void batch(const A& fun, T& accum, const L& mem, Args&&... args) {
472 std::vector<std::future<T>> futures; futures.reserve(mem.size());
473 for (auto& val : mem) {
474 futures.emplace_back(pool.submit_task([val = std::move(val), &fun, &args...]() {return init<T>(fun, val, args...);}));
475 }
476 for (auto& future : futures) extend(accum, future.get());
477 }
